## 🏗️ البنية المقترحة

### Component Structure

```typescript
// UnifiedSectionPage.tsx
interface UnifiedSectionPageProps {
  contentType: 'movies' | 'series' | 'anime' | 'gaming' | 'software'
  activeFilter?: 'all' | 'trending' | 'top-rated' | 'latest' | 'upcoming'
  genre?: string
  year?: number
}

const UnifiedSectionPage = ({ contentType, activeFilter = 'all', genre, year }: UnifiedSectionPageProps) => {
  // 1. Fetch ALL content for this section
  const allContent = useQuery(['all-content', contentType])
  
  // 2. Apply filters based on activeFilter, genre, year
  const filteredContent = useMemo(() => {
    let items = allContent.data || []
    
    if (activeFilter === 'trending') {
      items = items.filter(/* trending logic */)
    } else if (activeFilter === 'top-rated') {
      items = items.filter(/* top-rated logic */)
    }
    // ... more filters
    
    if (genre) {
      items = items.filter(item => item.genre === genre)
    }
    
    if (year) {
      items = items.filter(item => item.year === year)
    }
    
    return items
  }, [allContent.data, activeFilter, genre, year])
  
  return (
    <div>
      {/* NO Hero Cards */}
      
      {/* Filter Tabs */}
      <FilterTabs 
        active={activeFilter}
        options={['all', 'trending', 'top-rated', 'latest', 'upcoming']}
      />
      
      {/* Content Grid */}
      <ContentGrid items={filteredContent} />
      
      {/* Advanced Filters */}
      <AdvancedFilters 
        genre={genre}
        year={year}
        sortBy={sortBy}
      />
    </div>
  )
}
```

## 🔄 مثال على التدفق

### المستخدم يفتح `/movies`:
1. يرى روابط: [الكل] [الرائج] [الأعلى] [الأحدث] [قريباً]
2. يرى **كل** الأفلام في Grid
3. يمكنه الفلترة حسب التصنيف/السنة/التقييم
4. يمكنه الترتيب حسب الشعبية/التقييم/التاريخ

1. ينتقل إلى `/movies/trending`
2. يرى نفس الروابط لكن "الرائج" مميز
3. يرى **فقط** الأفلام الرائجة في Grid
4. نفس خيارات الفلترة والترتيب متاحة

### المستخدم يختار تصنيف "أكشن":
1. URL يتحدث إلى `/movies/trending?genre=action`
2. يرى **فقط** أفلام الأكشن الرائجة
3. يمكنه إضافة فلاتر أخرى (سنة، تقييم)
